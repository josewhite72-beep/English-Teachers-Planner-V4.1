// NUEVA FUNCIÓN (REEMPLAZAR CON ESTO):

const handleExportDocx = async () => {
  setExporting(true);
  try {
    const dataToExport = editMode && editedPlanner ? editedPlanner : generatedPlanner;
    
    // Llamar a la Vercel Function
    const response = await fetch('/api/export-docx', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(dataToExport),
    });

    if (!response.ok) {
      throw new Error(`Server error: ${response.status}`);
    }

    // Descargar el archivo
    const blob = await response.blob();
    const url = window.URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.setAttribute('download', `lesson_planner_${dataToExport.grade}_${dataToExport.scenario}.docx`);
    document.body.appendChild(link);
    link.click();
    link.remove();
    window.URL.revokeObjectURL(url);

    toast.success(language === 'es' ? 'Documento exportado exitosamente' : 'Document exported successfully');
  } catch (error) {
    console.error('Error exporting DOCX:', error);
    toast.error(language === 'es' ? 'Error al exportar documento' : 'Error exporting document');
  } finally {
    setExporting(false);
  }
};
